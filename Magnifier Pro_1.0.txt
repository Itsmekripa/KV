<script>
let data = { labels: [], en: [], de: [], fr: [], zh: [], ja: [], es: [], pt: [] };
let trans = {};
let glossary = [];
const LANGS = [['de','DE'],['fr','FR'],['zh','ZH'],['ja','JA'],['es','ES'],['pt','PT']];
const CJK = ['zh','ja'];

function esc(s) { return (s||'').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;'); }

function setMode(m) {
  document.body.classList.toggle('admin', m === 'admin');
  document.getElementById('modeTranslator').classList.toggle('active', m === 'translator');
  document.getElementById('modeAdmin').classList.toggle('active', m === 'admin');
  if (m === 'translator') {
    let active = document.querySelector('.tab.active');
    if (active && active.classList.contains('admin-only')) {
      document.querySelectorAll('.tab-content').forEach(t => t.classList.add('hidden'));
      document.getElementById('bulk').classList.remove('hidden');
      document.querySelectorAll('.tab').forEach(b => b.classList.remove('active'));
      document.querySelector('.tab').classList.add('active');
    }
  }
  botMsg(m === 'admin' ? '⚙️ Admin mode: Scan, Validator, DEX, Browser unlocked.' : '👤 Translator mode: focused everyday view.');
}

function switchTab(e, tab) {
  document.querySelectorAll('.tab-content').forEach(t => t.classList.add('hidden'));
  document.getElementById(tab).classList.remove('hidden');
  document.querySelectorAll('.tab').forEach(b => b.classList.remove('active'));
  e.target.classList.add('active');
}

function analyze() {
  let en = document.getElementById('en').value.trim();
  if (!en) { botMsg('⚠️ Paste English HTML first!'); return; }
  data = { labels: [], en: [], de: [], fr: [], zh: [], ja: [], es: [], pt: [] };
  trans = {};

  function parse(html) {
    let doc = new DOMParser().parseFromString(html, 'text/html');
    ['script','style'].forEach(t => doc.querySelectorAll(t).forEach(el => el.remove()));
    let labels = [];
    doc.querySelectorAll('h1,h2,h3,button,a,label,span,p').forEach(el => {
      let text = (el.textContent||'').trim().replace(/\s+/g,' ');
      if (text && text.length > 1 && text.length < 100) {
        labels.push({ text: text.substring(0,40), tag: el.tagName.toLowerCase(), risk: (el.id||'').match(/\d{6,}/) ? 'HIGH' : 'LOW' });
      }
    });
    return labels;
  }

  data.en = parse(en);
  ['de','fr','zh','ja','es','pt'].forEach(l => {
    let v = document.getElementById(l).value.trim();
    if (v) data[l] = parse(v);
  });

  let map = {};
  data.en.forEach(el => { if (!map[el.text]) map[el.text] = { ...el }; });
  data.labels = Object.values(map);

  ['de','fr','zh','ja','es','pt'].forEach(lang => {
    if (data[lang].length) {
      let lm = {};
      data[lang].forEach(el => { if (!lm[el.text]) lm[el.text] = el; });
      let ll = Object.values(lm);
      data.labels.forEach((lab, i) => {
        if (!trans[i]) trans[i] = { de:'', fr:'', zh:'', ja:'', es:'', pt:'' };
        if (ll[i]) trans[i][lang] = ll[i].text;
      });
    }
  });

  renderAll();
  botMsg('✅ ' + data.labels.length + ' labels analyzed. Open Workspace for side-by-side review.');
}

function expansion(src, tgt, lang) {
  if (!tgt) return null;
  let r = tgt.length / Math.max(src.length, 1);
  let pct = Math.round((r - 1) * 100);
  if (CJK.includes(lang)) return { pct, level: 'low' };
  if (r > 1.5) return { pct, level: 'high' };
  if (r > 1.3) return { pct, level: 'med' };
  return { pct, level: 'low' };
}

function countExpansionWarnings() {
  let n = 0;
  data.labels.forEach((l,i) => {
    LANGS.forEach(([k]) => {
      let e = expansion(l.text, (trans[i]||{})[k], k);
      if (e && e.level !== 'low') n++;
    });
  });
  return n;
}

function filledCount(lang) {
  let f = 0;
  data.labels.forEach((l,i) => { if (trans[i] && trans[i][lang]) f++; });
  return f;
}

function renderAll() {
  let total = data.labels.length;
  document.getElementById('total').textContent = total;
  document.getElementById('high').textContent = data.labels.filter(l=>l.risk==='HIGH').length;
  document.getElementById('langs').textContent = ['en','de','fr','zh','ja','es','pt'].filter(l=>data[l].length).length;
  document.getElementById('expwarn').textContent = countExpansionWarnings();

  let tt = '<tr><th>Language</th><th>Complete</th><th>Missing</th></tr>';
  LANGS.forEach(([k,n]) => {
    if (data[k].length) {
      let f = filledCount(k);
      let pct = total ? Math.round(f/total*100) : 0;
      let c = pct>=80?'low':pct>=50?'med':'high';
      tt += `<tr><td>${n}</td><td><span class="badge badge-${c}">${pct}%</span></td><td>${total-f}</td></tr>`;
    }
  });
  document.getElementById('transtable').innerHTML = tt;

  renderWorkspace(); renderMatrix(); renderScan(); renderValidator(); renderDex(); renderBrowser(); renderMissing(); renderGloss();
}

function renderWorkspace() {
  let lang = document.getElementById('wsLang').value;
  document.getElementById('wsBody').innerHTML = data.labels.map((l,i) => {
    let t = (trans[i]||{})[lang] || '';
    let e = expansion(l.text, t, lang);
    let badge = !t ? '<span class="badge badge-grey">EMPTY</span>'
      : `<span class="badge badge-${e.level}">${e.pct>0?'+':''}${e.pct}%</span>`;
    return `<tr><td>${i+1}</td><td><strong>${esc(l.text)}</strong></td>
      <td><input class="ws-input" data-idx="${i}" data-lang="${lang}" value="${esc(t)}"></td>
      <td>${badge}</td></tr>`;
  }).join('') || '<tr><td colspan="4" style="text-align:center;padding:40px;">No labels</td></tr>';

  document.querySelectorAll('#wsBody .ws-input').forEach(inp => {
    inp.addEventListener('change', function() {
      let i = +this.dataset.idx, lg = this.dataset.lang;

function renderValidator() {
  let total = data.labels.length;
  let issues = [];
  data.labels.forEach((l,i) => {
    let t = trans[i]||{};
    LANGS.forEach(([k,n]) => {
      if (data[k].length && !t[k]) issues.push({ label:l.text, issue:'Missing '+n, sev:'high' });
      if (t[k] && t[k] === l.text && !CJK.includes(k)) issues.push({ label:l.text, issue:n+' identical to EN (untranslated?)', sev:'med' });
      let e = expansion(l.text, t[k], k);
      if (e && e.level === 'high') issues.push({ label:l.text, issue:n+' expansion +'+e.pct+'% (breaks 50% budget)', sev:'high' });
      else if (e && e.level === 'med') issues.push({ label:l.text, issue:n+' expansion +'+e.pct+'% (over 30% budget)', sev:'med' });
    });
    glossary.forEach(g => {
      if (l.text.toLowerCase().includes(g.en.toLowerCase()) && t[g.lang] && !t[g.lang].toLowerCase().includes(g.term.toLowerCase())) {
        issues.push({ label:l.text, issue:'Glossary: '+g.lang.toUpperCase()+' must use "'+g.term+'" for "'+g.en+'"', sev:'high' });
      }
    });
  });
  let html = `<div class="stats"><div class="stat-card"><div class="stat-label">Checks</div><div class="stat-value">${total*8}</div></div><div class="stat-card"><div class="stat-label">Issues</div><div class="stat-value" style="color:${issues.length?'#dc2626':'#2EA44F'};">${issues.length}</div></div></div>`;
  if (issues.length) {
    html += '<table><tr><th>Label</th><th>Issue</th></tr>' + issues.slice(0,150).map(x => `<tr><td>${esc(x.label)}</td><td><span class="badge badge-${x.sev}">${esc(x.issue)}</span></td></tr>`).join('') + '</table>';
    if (issues.length > 150) html += `<p style="margin-top:6px;">+${issues.length-150} more (see Report export)</p>`;
  } else html += '<p>✅ All checks passed.</p>';
  document.getElementById('validatorBody').innerHTML = html;
  window._issues = issues;
}

function renderMissing() {
  let total = data.labels.length;
  let html = '';
  LANGS.forEach(([k,n]) => {
    if (!data[k].length) return;
    let miss = [];
    data.labels.forEach((l,i) => { if (!trans[i] || !trans[i][k]) miss.push(l.text); });
    let pct = total ? Math.round((total-miss.length)/total*100) : 0;
    let c = pct>=80?'low':pct>=50?'med':'high';
    html += `<div class="section-title">${n} — <span class="badge badge-${c}">${pct}%</span> (${miss.length} missing)</div>`;
    if (miss.length) html += '<table>' + miss.slice(0,50).map(t => `<tr><td>${esc(t)}</td></tr>`).join('') + '</table>' + (miss.length>50?`<p>+${miss.length-50} more in Missing CSV</p>`:'');
  });
  document.getElementById('missingBody').innerHTML = html || 'No non-English languages loaded.';
}

function download(content, filename, type) {
  let blob = new Blob(["\ufeff"+content], { type });
  let url = URL.createObjectURL(blob);
  let a = document.createElement('a');
  a.href = url; a.download = filename; a.click();
  URL.revokeObjectURL(url);
}
